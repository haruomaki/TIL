```haskell
{- HLINT ignore "Use =<<" -}
import System.IO (hFlush, stdout)
import Text.Read (readMaybe)

readInt :: IO Int
readInt = do
  putStr "数値を入力してください: "
  hFlush stdout
  line <- getLine
  case readMaybe line of
    Just n -> return n
    Nothing -> do
      putStrLn "整数以外は入力できません！"
      readInt

-- bindを用いてjoinを書く
join :: (Monad m) => m (m b) -> m b
join = flip (>>=) id

-- joinを用いてbindを書く
bind :: (Monad m) => m a -> (a -> m b) -> m b
bind = (join .) . flip fmap

main :: IO ()
main = do
  bind readInt print -- 普通のbind
  join (print <$> readInt) -- joinでbindを代替
```
